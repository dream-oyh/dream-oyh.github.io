def corr2d(X,K):
  K_h, K_w = K.shape
  X_h, X_w = X.shape
  Y = torch.zeros((X_h-K_h+1, X_w-K_w+1))
  for i in range(Y.shape[0]):
    for j in range(Y.shape[1]):
      Y[i,j] = (X[i:i+K_h,j:j+K_w]*K).sum()
  return Y

  def __init__(self, kernel_size):
