    def quick_sort(array: list):
    less = []
    greater = []
    equals = []
    array_length = len(array)
    if array_length <= 1:
        return array

    base_value = random.choice(array)
    for num in array:
        if num < base_value:
            less.append(num)
        elif num > base_value:
            greater.append(num)
        elif num == base_value:
            equals.append(num)
    sorted_less_array = quick_sort(less)
    sorted_great_array = quick_sort(greater)
    return sorted_less_array + equals + sorted_great_array
